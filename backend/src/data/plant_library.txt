CREATE TABLE authors (
id INT PRIMARY KEY AUTO_INCREMENT,
name VARCHAR(255),
email VARCHAR(255) UNIQUE,
bio VARCHAR(500),
created_at DATETIME
);

CREATE TABLE categories (
id INT PRIMARY KEY AUTO_INCREMENT,
name VARCHAR(255) UNIQUE,
description VARCHAR(255)
);

CREATE TABLE plants (
id INT PRIMARY KEY AUTO_INCREMENT,
name VARCHAR(255) UNIQUE,
img_url VARCHAR(255),
latin_name VARCHAR(255),
plant_family VARCHAR(255),
description TEXT,
care_instructions VARCHAR(500),
sun_requirement VARCHAR(255),
water_requirement VARCHAR(255),
soil_type VARCHAR(255),
blooming_season VARCHAR(255),
difficulty_level VARCHAR(255),
created_at DATETIME,
updated_at DATETIME
);

CREATE TABLE tags (
id INT PRIMARY KEY AUTO_INCREMENT,
name VARCHAR(255) UNIQUE
);

CREATE TABLE articles (
id INT PRIMARY KEY AUTO_INCREMENT,
author_id INT,
title VARCHAR(255),
slug VARCHAR(255) UNIQUE,
img_url VARCHAR(255),
content LONGTEXT,
excerpt TEXT,
category_id INT,
published_date DATETIME,
created_at DATETIME,
updated_at DATETIME,
is_premium BOOLEAN DEFAULT FALSE,
FOREIGN KEY (author_id) REFERENCES authors(id),
FOREIGN KEY (category_id) REFERENCES categories(id)
);

CREATE TABLE planting_schedule (
id INT PRIMARY KEY AUTO_INCREMENT,
plant_id INT,
planting_start_month INT,
planting_end_month INT,
harvest_start_month INT,
harvest_end_month INT,
notes VARCHAR(500),
FOREIGN KEY (plant_id) REFERENCES plants(id)
);

CREATE TABLE article_plants (
article_id INT,
plant_id INT,
PRIMARY KEY (article_id, plant_id),
FOREIGN KEY (article_id) REFERENCES articles(id) ON DELETE CASCADE,
FOREIGN KEY (plant_id) REFERENCES plants(id) ON DELETE CASCADE
);

CREATE TABLE article_tags (
article_id INT,
tag_id INT,
PRIMARY KEY (article_id, tag_id),
FOREIGN KEY (article_id) REFERENCES articles(id) ON DELETE CASCADE,
FOREIGN KEY (tag_id) REFERENCES tags(id) ON DELETE CASCADE
);

SELECT
      p.*,
      ps.id AS schedule_id,
      ps.planting_start_month,
      ps.planting_end_month,
      ps.harvest_start_month,
      ps.harvest_end_month,
      ps.notes
    FROM plants p
    LEFT JOIN planting_schedule ps ON ps.plant_id = p.id
    WHERE p.id = 3;

--

SELECT * FROM articles ORDER BY published_date DESC LIMIT 1;

--

SELECT a.*,
    authors.name AS author_name FROM articles a
    LEFT JOIN authors ON a.author_id = authors.id
    ORDER BY published_date DESC
    LIMIT 1;

--

SELECT name, latin_name, plant_family, sun_requirement, water_requirement, soil_type, care_instructions
FROM plants;

--

SELECT a.title, a.slug, c.name AS category, a.published_date
FROM articles a
JOIN authors au ON a.author_id = au.id
JOIN categories c ON a.category_id = c.id
WHERE au.name = 'Marcus Chill';

--

SELECT ar.title, ar.slug, au.name AS author
FROM articles ar
JOIN article_plants ap ON ar.id = ap.article_id
JOIN plants p ON ap.plant_id = p.id
JOIN authors au ON ar.author_id = au.id
WHERE p.name = 'Basil';

--

SELECT p.name, ps.planting_start_month, ps.planting_end_month, ps.harvest_start_month, ps.harvest_end_month, ps.notes
FROM planting_schedule ps
JOIN plants p ON ps.plant_id = p.id
ORDER BY ps.planting_start_month;

--

SELECT title, slug, published_date
FROM articles
WHERE is_premium = TRUE;

--

SELECT c.name AS category, COUNT(*) AS article_count
FROM articles a
JOIN categories c ON a.category_id = c.id
GROUP BY c.id;

--

SELECT ar.title, GROUP_CONCAT(t.name) AS tags
FROM articles ar
JOIN article_tags at ON ar.id = at.article_id
JOIN tags t ON at.tag_id = t.id
GROUP BY ar.id;

--

SELECT name, difficulty_level, sun_requirement, water_requirement
FROM plants
WHERE difficulty_level = 'Easy';

--

SELECT name, blooming_season, sun_requirement
FROM plants
WHERE blooming_season LIKE '%Summer%';
